import numpy as np

# Данные Роструда 2020-2025

disputes = [124,138,155,171,195,214]

plt.figure(figsize=(10,6))
plt.plot(years, disputes,
  color="#f4b942", linewidth=3)
plt.fill_between(years, disputes,
  alpha=0.15, color="#f4b942")
plt.savefig("disputes.png", dpi=150)
